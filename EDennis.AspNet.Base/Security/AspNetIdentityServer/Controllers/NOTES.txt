Features of AspNetIdentityServer
	1. Application Property of Role
	2. Organization Property of User
	3. UserClientClaims materialized view
	4. DomainIdentityProfileService
	5. Guid Keys for User and Role (with CombGuid default)
	6. Role-associated claims resolved by app
	

Users

	[HttpPost]
	Create([FromBody] TUser user)

	[HttpPatch("{id:Guid}")]
	Patch([FromRoute] Guid id, [FromBody] JsonElement jsonElement) 
	
	[HttpDelete("{id:Guid}")]
	Delete([FromRoute] Guid id)

	[HttpGet("{id:Guid}")]
	Get([FromRoute] Guid id)
	
	[HttpPut("{id:Guid}/roles/dict")]
	UpdateRoles([FromRoute] Guid id, [FromBody] Dictionary<string,string[]> roles)
	
	[HttpPut("{id:Guid}/roles/string-array")]	
	UpdateRoles([FromRoute] Guid id, [FromBody] string[] roles)

	[HttpPut("{id:Guid}/roles/obj-array")]	
	UpdateRoles([FromRoute] Guid id, [FromBody] ICollection<UserRole> roles)
	
	[HttpPut("{id:Guid}/claims/dict")]
	UpdateClaims([FromRoute] Guid id, [FromBody] Dictionary<string,string[]> claims)

	[HttpPut("{id:Guid}/claims/string-array")]
	UpdateClaims([FromRoute] Guid id, [FromBody] string[] claims)

	[HttpPut("{id:Guid}/claims/obj-array")]
	UpdateClaims([FromRoute] Guid id, [FromBody] ICollection<UserClaim> claims)

	[HttpGet("{id:Guid}/GetUserJsonWithDictClaimsAndRoles")]
	GetJson([FromRoute] Guid id)
	//di.GetUserJsonWithDictClaimsAndRoles (UserId)
	
	[HttpGet("{id:Guid}/GetUserJsonWithStringArrayClaimsAndRoles")]
	GetJson([FromRoute] Guid id)
	//di.GetUserJsonWithStringArrayClaimsAndRoles (UserId)
	
	[HttpGet("{id:Guid}/GetUserJsonWithObjArrayClaimsAndRoles")]
	GetJson([FromRoute] Guid id)
	//di.GetUserJsonWithObjArrayClaimsAndRoles (UserId)

	[HttpGet("app/{appName}")]
	GetApplicationUsers([FromRoute] string appName, int pageNumber, int pageSize)
	
	[HttpGet("org/{orgName}")]
	GetOrganizationUsers([FromRoute] string orgName, int pageNumber, int pageSize)
